class PipeInventory:
    def __init__(self):
        self.pipes = {}

    def add_pipe(self, pipe_type, diameter, quantity):
        if pipe_type in self.pipes:
            if diameter in self.pipes[pipe_type]:
                self.pipes[pipe_type][diameter] += quantity
            else:
                self.pipes[pipe_type][diameter] = quantity
        else:
            self.pipes[pipe_type] = {diameter: quantity}
